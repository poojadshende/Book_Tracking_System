create or replace PROCEDURE ADD_REQUEST 
(
  V_STUDENT_ID IN VARCHAR2 
, V_STUDENT_NAME IN VARCHAR2 
, V_BOOK_ID IN NUMBER 
, ERROR_MSG OUT VARCHAR2 
) AS 
v_ID NUMBER;
BEGIN
  v_ID:=seq_request_id.nextval();
  insert into REQUESTS(REQ_ID,STUDENT_NAME,BOOK_ID,REQ_DATE,REQ_STATUS,STUDENT_ID) values(v_ID,V_STUDENT_NAME,V_BOOK_ID,sysdate,'INCOMPLETE',V_STUDENT_ID);
  if SQL%ROWCOUNT >0 then
    ERROR_MSG:='001';
  else
    ERROR_MSG:='000';
  end if;
END ADD_REQUEST;
